 create database and write this on your PostgreSql Server(pgadmin4) in Query Tool:
 'CREATE TABLE users(
    id SERIAL PRIMARY KEY,
    user_id INTEGER UNIQUE,
    username TEXT
 );
 CREATE TABLE spends(
    amount INTEGER,
    user_id BIGINT,
    currency TEXT,
    create_date DATE,
    description TEXT,
    spend_id SERIAL PRIMARY KEY,
    FOREIGN KEY(user_id) REFERENCES Users(user_id)
 );
   CREATE VIEW numberedspends AS
   SELECT *,ROW_NUMBER() OVER (PARTITION BY user_id ORDER BY create_date) as number_of_expense
   from spends ;
'
